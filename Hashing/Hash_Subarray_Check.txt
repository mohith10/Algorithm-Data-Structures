#include<bits/stdc++.h>
typedef long long ll;

using namespace std;
int main()
{
    map<ll,ll> tab;
    int a[10],sub[5];
    int n=10;
    for(ll i=0;i<n;i++)
    {
        cin>>a[i];
    }
    bool res=true;
    for(ll i=0;i<5;i++)
       {
           cin>>sub[i];
           tab[sub[i]]++;
       }
    for(ll i=0;i<n;i++)
    {
        if(tab[a[i]]>0)
            tab[a[i]]--;
        else
        {
            res=false;
            break;
        }
    }
    if(res)
    cout<<"it is subarray";
    else cout<<"Not a subarray";
    
    
}
